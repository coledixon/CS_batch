using System;


namespace CreateAClassDrill
{
    public class Math
    {
        public int Addition(int x, int y)
        {
            return x + y;
        }

        public int Divide(int x, int y)
        {
            return x / y;
        }

        public int Multiply(int x, int y)
        {
            return x * y;
        }
    }
}
